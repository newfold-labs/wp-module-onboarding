595     * @param object $container The brand plugin container.
596     * @return string
597     */
598    public static function set_current_brand( $container ) {
599        if ( ! defined( 'NFD_ONBOARDING_PLUGIN_BRAND' ) ) {
600            $brand = $container->plugin()->brand;
601            if ( empty( $brand ) ) {
602                $brand = 'WordPress';
603            }
604
605            if ( false !== stripos( $brand, 'hostgator' ) ) {
606                $region = strtolower( $container->plugin()->region );
607                $brand  = "hostgator-{$region}";
608            }
609
610            $brand = sanitize_title_with_dashes( str_replace( '_', '-', $brand ) );
611            define( 'NFD_ONBOARDING_PLUGIN_BRAND', $brand );
612
613            return $brand;
614        }
